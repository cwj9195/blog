  • 闭包

  • 去重 Array.from(new Set([1,1,2,3]))

  • 闭包 函数内嵌套函数，并且函数执行完后，内部函数会被return返回引用，这样内部函数可以访问外部函数中定义的变量，于是就生成了一个闭包。

  • call apply bind 改变this的指向，用法不同

    • fn.apply(改变指向后对象, [1, 2]) //立即执行
    • fn.call(改变指向后对象, 1, 2) //立即执行
    • fn.bind(改变指向后对象, 1, 2)() //返回函数本身，不执行
      • fn.apply(this, arguments)
      • fn.call(this, ...arguments)

onMounted(() => {
  if (props.Type === 'default') handlerKeyupEnter()

  observer.value = new ResizeObserver((arg) => {
    const { width } = arg[0].contentRect
    let span = 6
    if (width <= 600) span = 12
    else if (width <= 900) span = 8
    else if (width <= 1200) span = 6
    else if (width >= 1800) span = 4
    defaultSpan.value = span
  })
  tcFormRef.value && observer.value.observe(tcFormRef.value)
})
onUnmounted(() => {
  tcFormRef.value && observer.value.unobserve(tcFormRef.value)
})
