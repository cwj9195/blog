    asyncPromise(array, poolLimit = 1, iteratorFn) {
        let i = 0;
        let ret = []; // 用来存储所有的Promise任务
        let executing = []; // 正在执行的Promise对象

        const enqueue = function () {
            if (i === array.length) { return Promise.resolve(); }       // 边界处理，array为空数组
            const item = array[i++];  // 获取item
            const p = Promise.resolve().then(() => iteratorFn(item, i, array)); // 初始化一个promise,执行一个返回Promise接口的方法（使用者的）
            const e = p.then(() => executing.splice(executing.indexOf(e), 1));      // 先走下面的executing.push  等promise执行完毕，再走executing数组中删除
            ret.push(p); executing.push(e);     // console.log({ ret, executing })
            let r = executing.length >= poolLimit ? Promise.race(executing) : Promise.resolve()     // 使用Promise.rece，每当executing数组中promise数量低于poolLimit，就实例化新的promise并执行
            return r.then(() => enqueue());  // r这个promise成功就接着递归，直到遍历完array
        };
        return enqueue().then(() => { return Promise.all(ret) }); // 最后呢，用Promise.all 等待全部任务完成，返回Promise的结果输出
    },

    dataURLtoBlob(dataurl) {
        let arr = dataurl.split(','),
            mime = arr[0].match(/:(.*?);/)[1],
            bstr = atob(arr[1]),
            n = bstr.length,
            u8arr = new Uint8Array(n);

        while (n--) { u8arr[n] = bstr.charCodeAt(n); }
        return new Blob([u8arr], { type: mime });
    },

            .then((res) => {
                updateManager.onUpdateReady(() => updateManager.applyUpdate())
                updateManager.onUpdateFailed(() => uni.showModal({ content: '新版本已经上线啦~，请您删除当前小程序，重新搜索打开', showCancel: false }))
            })
            .catch(err => console.log(`版本检测-无更新`, err))
    },

    uniGetSystemInfo() {
        let sys = uni.getSystemInfoSync()
            , MenuButton = uni.getMenuButtonBoundingClientRect() || { top: 48, bottom: 80 }
            , statusBarHeight = sys.statusBarHeight   //状态栏高度
            , navBarHeight = MenuButton.bottom + (MenuButton.top - statusBarHeight)   //这是总高含状态栏的
            ;
        return {
            statusBarHeight,
            navBarHeight,
            MenuButton,
            ratio: 750 / sys.screenWidth,    //1px像素==多少的rpx   比例,
            safeAreaBottom: sys.screenHeight - (sys.safeArea.bottom || sys.screenHeight), //安全区域 -底
            systemInfo: sys,
        }
    },

     * 获取系统权限并返回调用
     * 先获取授权，如果拒绝
     * @param {*} api 
     * @returns 
     */
    async getSystemPower(api, obj = {}) {
        let res
        if (api == 'userLocation') {
            if (!(await uni.authorize({ scope: 'scope.userLocation' }))[1]) {
                if (!(await uni.getSetting())[1].authSetting['scope.userLocation']) {
                    await uni.showModal({ content: '检测到您没打开获取位置功能权限', showCancel: false })
                    uni.openSetting()
                }
            }
            res = (await uni.getLocation({ geocode: true }))[1]
        }
        if (api == 'writePhotosAlbum') {
            if (!(await uni.authorize({ scope: 'scope.writePhotosAlbum' }))[1]) {
                if (!(await uni.getSetting())[1].authSetting['scope.writePhotosAlbum']) {
                    await uni.showModal({ content: '检测到您没打开照片保存功能', showCancel: false })
                    uni.openSetting()
                }
            }
            res = (await uni.saveImageToPhotosAlbum({ filePath: obj.filePath }))[1]
        }
        return res
    },

    keepAlive(type, keepAlive, cptName) {
        keepAlive = JSON.parse(JSON.stringify(keepAlive))
        // 缓存-去重添加
        if (type == 'add') {
            return Array.from(new Set(keepAlive.concat(cptName)))
        }
        // 缓存-删除
        if (type == 'del') {
            let i = keepAlive.indexOf(cptName)
            keepAlive.splice(i, i > -1 ? 1 : 0)
            return Array.from(new Set(keepAlive))
        }
    },
